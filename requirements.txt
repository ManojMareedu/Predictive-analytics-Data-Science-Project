# The block below is pinned to the EXACT versions that wrote
# exported_model/model.pkl, and must stay in step with
# exported_model/requirements.txt (MLflow regenerates that file on every export).

# These cannot float. A pickled sklearn Pipeline only loads under the version that
# created it, and Streamlit Community Cloud installs *this file and nothing else* --
# there is no second install step to correct a bad resolution the way the Dockerfile
# has. Left as `scikit-learn>=1.3`, a fresh Cloud build picks up whatever is current
# and the dashboard dies on load with
# `AttributeError: Can't get attribute '_RemainderColsList'`.

pandas==2.3.3
cloudpickle==3.1.2
mlflow==3.1.4

# Application layer: not part of the pickle, so bounded to a major version rather
# than pinned. Upgrades here cannot break model loading.
fastapi>=0.110,<1.0
uvicorn[standard]>=0.27,<1.0
pydantic>=2.5,<3.0

# st.plotly_chart/st.dataframe/st.image, which older releases reject. Cloud
# resolves to the newest release in range, but the floor should still be a
# version the app actually runs on.
streamlit>=1.50,<2.0
